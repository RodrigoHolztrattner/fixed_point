83  template<class Rep>
84  constexpr Rep sqrt_solve3(
85  Rep n,
86  Rep bit,
87  Rep result)
88  {
89  return (bit!=Rep{0})
90  ? (n>=result+bit)
91  ? sqrt_solve3<Rep>(
92  static_cast<Rep>(n-(result+bit)),
93  bit >> 2,
94  static_cast<Rep>((result >> 1)+bit))
95  : sqrt_solve3<Rep>(n, bit >> 2, result >> 1)
96  : result;
97  }
